package com.kh.part02_scheduling.thread;

public class Tank implements Runnable{

	@Override
	public void run() {		
		for(int i = 0; i < 100; i++) {
		System.out.println(i + "Tank shooting...");
		try {
			Thread.sleep(100);
		} catch (InterruptedException e) {
			e.printStackTrace();
		}
	}
	
		
	}
	

}
